const greatExpectations = require('great-expectations-node');
const csv = require('csv');

const expectations = {
  "column_values_to_be_between": {
    "column": "passenger_count",
    "min_value": 1,
    "max_value": 6,
  },
};

const csvData = csv.parse('passenger_data.csv');

const results = greatExpectations.validate(expectations, csvData);

console.log(results);
